    function checkAuthentication() {
      if (!isAuthenticated) {
        const code = prompt('🔐 Enter admin access code to proceed:');
        if (code === 'niranjana') {
          isAuthenticated = true;
          showAlert('✅ Authentication successful!', 'success');
        } else {
          showAlert('❌ Incorrect access code. Access denied.', 'error');
          return false;
        }
      }
      return true;
    }

    function addStatusLog(message, type = 'info') {
      const log = document.getElementById('statusLog');

      entry.innerHTML = `[${timestamp}] ${message}`;

    }

    window.handleFileSelect = function(event) {
      const file = event.target.files[0];
      if (file) {
        handleFile(file);
      }
    };

    function handleFile(file) {
      if (file.type !== 'application/json') {
        showAlert('⚠️ Please select a valid JSON file.', 'warning');
        return;
      }

      const reader = new FileReader();
      reader.onload = function(e) {
        try {
          jsonData = JSON.parse(e.target.result);
          displayFileInfo(file);
          previewData();
        } catch (error) {
          showAlert('❌ Invalid JSON file format.', 'error');
          console.error('JSON Parse Error:', error);
        }
      };
      reader.readAsText(file);
    }

    function displayFileInfo(file) {
      document.getElementById('fileName').textContent = file.name;
      document.getElementById('fileDetails').textContent = `Size: ${(file.size / 1024).toFixed(2)} KB | Modified: ${new Date(file.lastModified).toLocaleDateString()}`;
      document.getElementById('fileInfo').classList.remove('hidden');
    }

    function previewData() {
      if (!jsonData) return;

      let collections = [];
      let totalRecords = 0;
      let sampleRecord = null;

      // Handle different JSON structures
      if (Array.isArray(jsonData)) {
        // Single collection export
        collections.push('Array Data');
        totalRecords = jsonData.length;
        sampleRecord = jsonData[0];
      } else if (typeof jsonData === 'object') {
        // Multi-collection export
        for (const [key, value] of Object.entries(jsonData)) {
          if (Array.isArray(value)) {
            collections.push(`${key} (${value.length} records)`);
            totalRecords += value.length;
            if (!sampleRecord && value.length > 0) {
              sampleRecord = { collection: key, data: value[0] };
            }
          }
        }
      }

      // Update UI
      document.getElementById('recordCount').textContent = totalRecords;
      
      const collectionsDiv = document.getElementById('collectionsFound');
      collectionsDiv.innerHTML = collections.map(c => `
• ${c}
`).join('');
      
      const sampleDiv = document.getElementById('sampleData');
      sampleDiv.textContent = JSON.stringify(sampleRecord, null, 2);
      
      document.getElementById('previewSection').classList.remove('hidden');
      showAlert(`📊 Data loaded: ${totalRecords} records in ${collections.length} collection(s)`, 'success');
    }

    window.startImport = async function() {
      if (!checkAuthentication()) return;
      
      const targetCollection = document.getElementById('targetCollection').value;
      const importMode = document.querySelector('input[name="importMode"]:checked').value;
      const batchSize = parseInt(document.getElementById('batchSize').value);

      if (!jsonData) {
        showAlert('❌ Please select a JSON file first.', 'error');
        return;
      }

      if (!targetCollection) {
        showAlert('❌ Please select a target collection.', 'error');
        return;
      }

      // Confirmation for dangerous operations
      if (importMode === 'replace') {
        const confirm = prompt(`⚠️ DANGER: This will DELETE ALL existing data in "${targetCollection}" collection.\nType "DELETE ALL" to confirm:`);
        if (confirm !== 'DELETE ALL') {
          showAlert('❌ Import cancelled for safety.', 'warning');
          return;
        }
      }

      // Reset stats
      importStats = { total: 0, successful: 0, skipped: 0, errors: 0, errorDetails: [] };
      
      // Show progress section
      document.getElementById('progressSection').classList.remove('hidden');
      document.getElementById('resultsSection').classList.add('hidden');
      document.getElementById('statusLog').innerHTML = '
Starting import process...
';

      try {
        await performImport(targetCollection, importMode, batchSize);
      } catch (error) {
        console.error('Import failed:', error);
        addStatusLog(`❌ Import failed: ${error.message}`, 'error');
        showAlert('❌ Import process failed.', 'error');
      }
    };

    async function performImport(targetCollection, importMode, batchSize) {
      addStatusLog(`🚀 Starting import to "${targetCollection}" collection`, 'info');
      addStatusLog(`📋 Mode: ${importMode}, Batch Size: ${batchSize}`, 'info');

      // Prepare data
      let recordsToImport = [];
      
      if (Array.isArray(jsonData)) {
        recordsToImport = jsonData;
      } else if (jsonData[targetCollection]) {
        recordsToImport = jsonData[targetCollection];
      } else {
        throw new Error(`No data found for collection "${targetCollection}"`);
      }

      importStats.total = recordsToImport.length;
      addStatusLog(`📊 Found ${importStats.total} records to import`, 'info');

      // Clear collection if replace mode
      if (importMode === 'replace') {
        await clearCollection(targetCollection);
      }

      // Process in batches
      const totalBatches = Math.ceil(recordsToImport.length / batchSize);
      
      for (let i = 0; i < totalBatches; i++) {
        const batchStart = i * batchSize;
        const batchEnd = Math.min(batchStart + batchSize, recordsToImport.length);
        const batch = recordsToImport.slice(batchStart, batchEnd);
        
        addStatusLog(`📦 Processing batch ${i + 1}/${totalBatches} (${batch.length} records)`, 'info');
        
        await processBatch(targetCollection, batch, importMode, i + 1, totalBatches);
        
        // Update progress
        const overallProgress = Math.round(((i + 1) / totalBatches) * 100);
        document.getElementById('overallProgress').textContent = `${overallProgress}%`;
        document.getElementById('overallProgressBar').style.width = `${overallProgress}%`;
        document.getElementById('batchProgress').textContent = `Batch ${i + 1} of ${totalBatches}`;
        
        // Small delay between batches
        await new Promise(resolve => setTimeout(resolve, 500));
      }

      // Show results
      showResults();
      addStatusLog(`✅ Import completed! ${importStats.successful} successful, ${importStats.errors} errors`, 'success');
      showAlert(`🎉 Import completed! ${importStats.successful}/${importStats.total} records imported successfully.`, 'success');
    }

    async function clearCollection(collectionName) {
      addStatusLog(`🗑️ Clearing existing data from "${collectionName}"...`, 'warning');
      
      try {
        const snapshot = await getDocs(collection(db, collectionName));
        const batch = writeBatch(db);
        
        snapshot.docs.forEach((doc) => {
          batch.delete(doc.ref);
        });
        
        await batch.commit();
        addStatusLog(`✅ Cleared ${snapshot.docs.length} existing records`, 'success');
      } catch (error) {
        addStatusLog(`❌ Error clearing collection: ${error.message}`, 'error');
        throw error;
      }
    }

    async function processBatch(collectionName, records, mode, batchNum, totalBatches) {
      const batch = writeBatch(db);
      const batchErrors = [];
      
      for (let i = 0; i < records.length; i++) {
        try {
          const record = records[i];
          const docId = record.id || `import_${Date.now()}_${i}`;
          
          // Remove the id field from data if it exists
          const docData = { ...record };
          delete docData.id;
          
          // Check if document exists for append mode
          if (mode === 'append') {
            const docRef = doc(db, collectionName, docId);
            const existingDoc = await getDocs(collection(db, collectionName));
            const exists = existingDoc.docs.some(d => d.id === docId);
            
            if (exists) {
              importStats.skipped++;
              continue;
            }
          }
          
          const docRef = doc(db, collectionName, docId);
          batch.set(docRef, docData, { merge: mode === 'merge' });
          
        } catch (error) {
          batchErrors.push(`Record ${i}: ${error.message}`);
          importStats.errors++;
        }
      }
      
      try {
        await batch.commit();
        const successfulInBatch = records.length - batchErrors.length;
        importStats.successful += successfulInBatch;
        
        addStatusLog(`✅ Batch ${batchNum} completed: ${successfulInBatch} successful`, 'success');
        
        if (batchErrors.length > 0) {
          addStatusLog(`⚠️ Batch ${batchNum} had ${batchErrors.length} errors`, 'warning');
          importStats.errorDetails.push(...batchErrors);
        }
        
      } catch (error) {
        addStatusLog(`❌ Batch ${batchNum} failed: ${error.message}`, 'error');
        importStats.errors += records.length;
        importStats.errorDetails.push(`Batch ${batchNum}: ${error.message}`);
      }
      
      // Update batch progress
      const batchProgress = Math.round((batchNum / totalBatches) * 100);
      document.getElementById('batchProgressBar').style.width = `${batchProgress}%`;
    }
